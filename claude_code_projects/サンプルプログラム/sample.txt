/**
 * 数学: 2点間の距離（ユークリッド距離）
 * @param {number} x1
 * @param {number} y1
 * @param {number} x2
 * @param {number} y2
 * @returns {number}
 */
function distance2D(x1, y1, x2, y2) {
  const dx = x2 - x1;
  const dy = y2 - y1;
  return Math.sqrt(dx * dx + dy * dy);
}

/**
 * 数学: 配列の平均値
 * @param {number[]} nums
 * @returns {number}
 */
function average(nums) {
  if (!Array.isArray(nums) || nums.length === 0) return 0;
  const sum = nums.reduce((acc, n) => acc + n, 0);
  return sum / nums.length;
}

/**
 * 文字列: 先頭を大文字にする（簡易）
 * @param {string} s
 * @returns {string}
 */
function capitalize(s) {
  if (typeof s !== 'string' || s.length === 0) return '';
  return s[0].toUpperCase() + s.slice(1);
}

/**
 * 文字列: 指定文字数で省略（末尾に "..."）
 * @param {string} s
 * @param {number} maxLen
 * @returns {string}
 */
function truncate(s, maxLen) {
  if (typeof s !== 'string') return '';
  if (typeof maxLen !== 'number' || maxLen <= 0) return '';
  if (s.length <= maxLen) return s;
  if (maxLen <= 3) return '.'.repeat(maxLen); // 例外的に短すぎる場合
  return s.slice(0, maxLen - 3) + '...';
}

module.exports = {
  distance2D,
  average,
  capitalize,
  truncate,
};

function executeLogicFlow(isPremium, spent) {
  console.log('=== Logic Flow Execution ===\n');

  // Start
  console.log('▶ Start\n');

  // Get user data
  console.log('▶ Get user data');
  console.log(`Input: isPremium=${isPremium}, spent=${spent}\n`);

  console.log(`▶ isPremium?`);

  let bonus = 0;

  if (isPremium) {
    console.log('▶ Yes → Check Spent ≥ 5000?');
    console.log(`Spent amount: ${spent}`);

    if (spent >= 5000) {
      // Yes: Both conditions met
      console.log('▶ Yes → Set bonus bonus = 1000');
      bonus = 1000;
    } else {
      // No: Premium but low spend
      console.log('▶ No → bonus = 0');
      bonus = 0;
    }
  } else {
    // No: Not premium
    console.log('▶ No → bonus = 0');
    bonus = 0;
  }

  console.log(`\n=== Result ===`);
  console.log(`Premium: ${isPremium}`);
  console.log(`Spent: ${spent}`);
  console.log(`Bonus: ${bonus}`);
  console.log('▶ End\n');
  console.log('─'.repeat(50) + '\n');
}

var fs = require('fs');

// コールバック地獄とvarの多用、古い関数の書き方
function processUserData(userId, callback) {
    var result = {};
    
    // 1. ユーザー情報の取得（擬似的な非同期処理）
    setTimeout(function() {
        // ダミーのデータ
        const data = [
            { id: 1, name: "Taro", type: "standard" },
            { id: 2, name: "Hanako", type: "premium" },
            { id: 3, name: "Sachiko", type: "standard" },
        ]
        var user = data[userId];
        result.user = user;
        
        // 2. 注文履歴の取得
        setTimeout(function() {
            var orders = [
                { id: "o1", amount: 1000 },
                { id: "o2", amount: 2500 }
            ];
            
            // varを使った古典的なループ処理
            var totalAmount = 0;
            for (var i = 0; i < orders.length; i++) {
                totalAmount = totalAmount + orders[i].amount;
            }
            result.orders = orders;
            result.totalAmount = totalAmount;
            
            // 3. プレミアムユーザーなら割引を適用
            if (user.type === "premium") {
                result.discountedTotal = totalAmount * 0.9;
            } else {
                result.discountedTotal = totalAmount;
            }
            
            // 4. 結果をファイルに保存
            var logText = "User: " + user.name + ", Total: " + result.discountedTotal;
            fs.writeFile('user-log.txt', logText, function(err) {
                if (err) {
                    console.log("Error writing file!");
                    callback(err, null);
                } else {
                    console.log("File written successfully.");
                    // 5. 最後にコールバックで結果を返す
                    callback(null, result);
                }
            });
            
        }, 500); // 注文取得に500ms
    }, 500); // ユーザー取得に500ms
}

// 実行部分
console.log("Processing started...");
const id = 2; // ユーザーIDを指定
processUserData(id, function(err, data) {
    if (err) {
        console.error("Failed:", err);
    } else {
        console.log("Success! Final Data:", data);
    }
});

import { Command } from 'commander';
import fs from 'fs';
import path from 'path';
import { fileURLToPath } from 'url';

const __dirname = path.dirname(fileURLToPath(import.meta.url));
const TODOS_FILE = path.join(__dirname, 'todos.json');

const program = new Command();

// ヘルパー関数：todos.jsonを読み込む
function loadTodos() {
  try {
    if (fs.existsSync(TODOS_FILE)) {
      const data = fs.readFileSync(TODOS_FILE, 'utf-8');
      return JSON.parse(data);
    }
  } catch (error) {
    console.error('Error reading todos file:', error.message);
  }
  return [];
}

// ヘルパー関数：todos.jsonに保存する
function saveTodos(todos) {
  try {
    fs.writeFileSync(TODOS_FILE, JSON.stringify(todos, null, 2));
  } catch (error) {
    console.error('Error saving todos file:', error.message);
  }
}

// タスク追加コマンド
program
  .command('add <description>')
  .description('Add a new todo task')
  .action((description) => {
    const todos = loadTodos();
    const newTodo = {
      id: todos.length > 0 ? Math.max(...todos.map(t => t.id)) + 1 : 1,
      description,
      completed: false,
      createdAt: new Date().toISOString()
    };
    todos.push(newTodo);
    saveTodos(todos);
    console.log(`✓ Task added (ID: ${newTodo.id})`);
  });

// タスク一覧表示コマンド

async function resizeImages(inputDir) {

    const outputDir = path.join(inputDir, 'output');
    await fs.mkdir(outputDir, { recursive: true });

    const files = (await fs.readdir(inputDir)).filter(f => /\.(jpg|png)$/i.test(f));

        await sharp(path.join(inputDir, file))
            .resize(800)
            .toFile(path.join(outputDir, file));
    }));

    spinner.succeed(`${files.length}枚の画像を処理しました！`);
}

import pc from 'picocolors';

// ヘルプ表示
function showHelp() {
  console.log(`
URL Health Monitor - picocolorsを使った死活監視CLI

使用方法:
  node monitor.js <URL> [options]

オプション:
  -i, --interval <ms>  監視間隔（ミリ秒）デフォルト: 5000
  -h, --help          このヘルプを表示

例:
  node monitor.js https://google.com
  node monitor.js https://example.com -i 3000
  `);

// タイムスタンプをグレーで返す
function getTimestamp() {
  const now = new Date();
  const hours = String(now.getHours()).padStart(2, '0');
  const minutes = String(now.getMinutes()).padStart(2, '0');
  const seconds = String(now.getSeconds()).padStart(2, '0');
  const ms = String(now.getMilliseconds()).padStart(3, '0');
  return pc.gray(`[${hours}:${minutes}:${seconds}.${ms}]`);
}

// ステータスに応じた色付けされたメッセージを返す
function formatStatus(statusCode, statusText) {
  if (statusCode === 200) {
    return pc.green(`✓ OK`);
  } else if (statusCode === 404) {
    return pc.yellow(`⚠ NOT FOUND`);
  } else if (statusCode >= 500) {
    return pc.bgRed(pc.white(` ✕ CRITICAL `));
  } else if (statusCode >= 400) {
    return pc.yellow(`⚠ ${statusCode} ${statusText}`);
  } else if (statusCode >= 300) {
    return pc.blue(`→ ${statusCode} ${statusText}`);
  } else {
    return pc.green(`✓ ${statusCode} ${statusText}`);
  }
}

// URLのヘルスチェック
async function checkHealth(url) {
  try {
    const response = await fetch(url, {
      method: 'HEAD',
      timeout: 10000,
      redirect: 'follow'
    });

    const statusCode = response.status;
    const statusText = response.statusText || 'No Status Text';

    console.log(
      `${getTimestamp()} ${url} ${formatStatus(statusCode, statusText)}`
    );
  } catch (error) {
    console.log(
      `${getTimestamp()} ${url} ${pc.bgRed(pc.white(` ✕ ERROR `))} ${pc.red(error.message)}`
    );
  }
}

// メイン処理
async function main() {
  const args = process.argv.slice(2);

  if (args.length === 0 || args.includes('-h') || args.includes('--help')) {
    showHelp();
    process.exit(0);
  }

  let url = args[0];
  let interval = 5000;

  // URLのバリデーション
  if (!url.startsWith('http://') && !url.startsWith('https://')) {
    url = 'https://' + url;
  }

  // オプションの解析
  for (let i = 1; i < args.length; i++) {
    if ((args[i] === '-i' || args[i] === '--interval') && args[i + 1]) {
      interval = parseInt(args[i + 1], 10);
      i++;
    }
  }

  console.log(pc.cyan(`\n🔍 URL Health Monitor Started\n`));
  console.log(pc.gray(`監視対象: ${url}`));
  console.log(pc.gray(`監視間隔: ${interval}ms\n`));

  // 初回チェック
  await checkHealth(url);

  // 定期的なチェック
  setInterval(() => {
    checkHealth(url);
  }, interval);
}

main();

  // バリデーション2：message が空または5文字以下か確認
  if (!message || message.trim().length <= 5) {
    // message が空または5文字以下の場合、400エラーを返す
    return res.status(400).json({
      error: 'メッセージは5文字を超える必要があります'
    });
  }

  // 全てのバリデーションに通った場合、受け取った内容をコンソールに表示
  console.log('========== お問い合わせフォーム受信 ==========');
  console.log(`メールアドレス: ${email}`);
  console.log(`メッセージ: ${message}`);
  console.log(`受信日時: ${new Date().toLocaleString('ja-JP')}`);
  console.log('==========================================');

  // ステータスコード 200（OK/成功）を返す
  res.status(200).json({
    message: 'お問い合わせをお受けしました。ご連絡ありがとうございます。',
    received: {
      email: email,
      message: message,
      timestamp: new Date().toLocaleString('ja-JP')
    }
  });
});

  posts.push(newPost);

  // Redirect to board

import { useState, useEffect } from 'react'

function App() {
  // ステート管理: カウンター
  const [count, setCount] = useState(0)

  // ステート管理: ユーザー一覧
  const [users, setUsers] = useState([])
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState(null)

  // useEffect: 初回描画時にAPIからユーザー一覧を取得
  useEffect(() => {
    const fetchUsers = async () => {
      try {
        setLoading(true)

        if (!response.ok) {
          throw new Error('データ取得に失敗しました')
        }

        setUsers(data)
        setError(null)

        setError(err.message)
      } finally {
        setLoading(false)
      }
    }

    fetchUsers()
  }, [])

  return (
    <>
      <section id="center">
        <h1>JSXサンプル</h1>
      </section>

      {/* ユーザー一覧 */}
      <section id="users" style={{ padding: '2rem', maxWidth: '800px', margin: '0 auto' }}>

            エラー: {error}
          </p>
        )}

        {/* ユーザーリストの表示 */}

import { useState, useEffect } from 'react'
import './App.css'

// 独自フックの定義
function useWindowSize() {
  const [width, setWidth] = useState(window.innerWidth);

  useEffect(() => {
    const handleResize = () => setWidth(window.innerWidth);
    window.addEventListener('resize', handleResize);
    // クリーンアップ処理（メモリ漏れ防止）
    return () => window.removeEventListener('resize', handleResize);
  }, []);

  return width;
}

// コンポーネントでの使用
function App() {
  const width = useWindowSize(); // 自作フックを呼び出すだけ！

  return (
    <>
      <section id="center">
        <h1>JSXサンプル</h1>
      </section>

      {/* ウィンドウ幅表示 */}
      <section id="users" style={{ 
        padding: '2rem', maxWidth: '800px', margin: '0 auto'
      }}>
        <h2>ウィンドウ幅</h2>

        <h3>現在の幅: {width}px - {width < 768 ? 'スマホ' : 'PC'}</h3>
      </section>
    </>
  )
}

import { Server } from "@modelcontextprotocol/sdk/server/index.js";
import { StdioServerTransport } from "@modelcontextprotocol/sdk/server/stdio.js";
import { ListToolsRequestSchema, CallToolRequestSchema } from "@modelcontextprotocol/sdk/types.js";

// ==================================================================
// MCPサーバーのインスタンスを作成します
// 第1引数: サーバーの情報（名前とバージョン）
// ==================================================================
const server = new Server({
  name: "add_numbers_server",   // このサーバーの名前
  version: "1.0.0",             // このサーバーのバージョン
}, {
  capabilities: {
    tools: {},                  // このサーバーはツール機能をサポートする
  },
});

// ツールの定義：提供するツール（機能）を事前に登録します
// ==================================================================
// SDK提供のスキーマオブジェクトを使用してハンドラを定義
server.setRequestHandler(ListToolsRequestSchema, async () => {
  return {
    // tools は、このサーバーが提供できるツールの配列です
    tools: [
      {
        name: "add_numbers",                         // ツール名
        description: "2つの数値を加算して結果を返します", // ツールの説明
        inputSchema: {
          type: "object",                 // 入力はオブジェクト型です
          properties: {
            a: {
              type: "number",             // 第1引数: 数値型
              description: "最初の数値",
            },
            b: {
              type: "number",             // 第2引数: 数値型
              description: "2番目の数値",
            },
          },
          required: ["a", "b"],           // "a" と "b" は両方必須です
        },
      },
    ],
  };
});

// ==================================================================
// ツール実行ハンドラ：クライアントがツールを呼び出したときの処理
// ==================================================================
// SDK提供のスキーマオブジェクトを使用してハンドラを定義
server.setRequestHandler(CallToolRequestSchema, async (request) => {
  // request.params.name: 実行されるツールの名前
  // request.params.arguments: ツールに渡される引数（オブジェクト）

  if (request.params.name === "add_numbers") {
    // クライアントが "add_numbers" ツールを呼び出した場合の処理

    // 引数から a と b を取得します
    const a = request.params.arguments.a;
    const b = request.params.arguments.b;

    // 2つの数値を加算します
    const result = a + b;

    // 結果をクライアントに返します
    return {
      content: [
        {
          type: "text",                       // 結果はテキスト形式
          text: `${a} + ${b} = ${result}`,    // 計算結果を含むテキスト
        },
      ],
    };
  }

  // 定義されていないツールが呼ばれた場合、エラーを返します
  throw new Error(`不明なツール: ${request.params.name}`);

// サーバーが起動したことをログに出力します
console.error("add_numbers MCPサーバーが起動しました");

// MCPサーバーをテストするクライアントスクリプト
// Node.jsの child_process モジュールを使ってサーバープロセスを起動し、
// JSON-RPCプロトコルで通信します

import { spawn } from "child_process";

// MCPサーバープロセスを起動
const serverProcess = spawn("node", ["index.js"], {
  stdio: ["pipe", "pipe", "pipe"], // stdin, stdout, stderr をパイプ化
});

let messageId = 0;

/**
 * MCPサーバーにJSON-RPC形式のリクエストを送信し、レスポンスを待つ関数
 * @param {string} method - RPC メソッド名（例：'tools/list'）
 * @param {object} params - メソッドのパラメータ
 */
function sendRequest(method, params = {}) {
  return new Promise((resolve, reject) => {
    messageId++;

    const request = {
      jsonrpc: "2.0",
      id: messageId,
      method: method,
      params:params,
    };

    console.error(`\n📤 送信するリクエスト (ID: ${messageId}):`);
    console.error(JSON.stringify(request, null, 2));

    // レスポンスを受け取るためのハンドラを1回限りで設定
    const responseHandler = (data) => {
      try {
        const response = JSON.parse(data.toString());

        // 受け取ったレスポンスのIDが一致するかチェック
        if (response.id === messageId) {
          console.error(`\n📥 受け取ったレスポンス:`);
          console.error(JSON.stringify(response, null, 2));

          serverProcess.stdout.removeListener("data", responseHandler);
          resolve(response);
        }
      } catch (err) {
        console.error("レスポンスのパース失敗:", err.message);
        reject(err);
      }
    };

    // stdoutのデータイベントをリッスン
    serverProcess.stdout.on("data", responseHandler);

    // リクエストをサーバーの stdin に送信
    serverProcess.stdin.write(JSON.stringify(request) + "\n");

    // タイムアウト設定（5秒）
    setTimeout(() => {
      serverProcess.stdout.removeListener("data", responseHandler);
      reject(new Error("リクエストタイムアウト"));
    }, 5000);
  });
}

// エラーハンドリング
serverProcess.stderr.on("data", (data) => {
  console.error(`\n📋 サーバーログ: ${data.toString()}`);
});

// メイン実行
async function main() {
  try {
    console.error("=".repeat(60));
    console.error("MCPサーバークライアント テスト開始");
    console.error("=".repeat(60));

    // 1. 利用可能なツールの一覧を取得
    console.error("\n【ステップ1】サーバーで利用可能なツール一覧を取得");
    await sendRequest("tools/list");

    // 2. 実際にツールを実行 (add_numbers を呼び出し)
    console.error("\n\n【ステップ2】add_numbers ツールを実行 (5 + 3)");
    await sendRequest("tools/call", {
      name: "add_numbers",
      arguments: { "a": 5, "b": 3 },
    });

    // 3. 別の値でツールを実行
    console.error("\n\n【ステップ3】add_numbers ツールを実行 (100 + 250)");
    await sendRequest("tools/call", {
      name: "add_numbers",
      arguments: { "a": 100, "b": 250 },
    });

    console.error("\n\n" + "=".repeat(60));
    console.error("✅ すべてのテストが完了しました");
    console.error("=".repeat(60));

    // サーバープロセスを終了
    serverProcess.kill();
    process.exit(0);
  } catch (error) {
    console.error("\n❌ エラーが発生しました:", error.message);
    serverProcess.kill();
    process.exit(1);
  }
}

// スクリプトを実行
main();

const NOTES_FILE = path.join(__dirname, "notes.json");

// ノートファイルを読み込む（同期処理）
function readNotes() {
  try {
    const data = fs.readFileSync(NOTES_FILE, "utf-8");
    return JSON.parse(data);
  } catch (error) {
    // ファイルが存在しない場合は空配列を返す
    if (error.code === "ENOENT") {
      return [];
    }
    throw error;
  }
}

// ノートファイルに書き込む（同期処理）
function writeNotes(notes) {
  fs.writeFileSync(NOTES_FILE, JSON.stringify(notes, null, 2), "utf-8");
}

// ===============================================================
// ツール実行ハンドラ：クライアントがツールを呼び出したときの処理
// ===============================================================
// SDK提供のスキーマオブジェクトを使用してハンドラを定義
server.setRequestHandler(CallToolRequestSchema, async (request) => {

  // request.params.arguments: ツールに渡される引数（オブジェクト）

  if (request.params.name === "add_numbers") {

  }

  // ===============================================================
  // get_system_info ツール実行処理：システム情報を取得して返す
  // ===============================================================
  if (request.params.name === "get_system_info") {
    // ...説明済みのため省略...
  }

  // ==================================================================
  // add_note ツール実行処理：メモを追加
  // ==================================================================
  if (request.params.name === "add_note") {
    const { content } = request.params.arguments;

    // 既存のメモを読み込む
    const notes = readNotes();

    // 新しいメモを作成して配列に追加
    const maxId = notes.length > 0 ? Math.max(...notes.map(n => n.id)) : 0;
    const newNote = {
      id: maxId + 1,
      createdAt: new Date().toISOString(),
      content: content,
    };
    notes.push(newNote);

    // ファイルに書き込む
    writeNotes(notes);

    return {
      content: [
        {
          type: "text",
          text: `メモを追加しました。\nID: ${newNote.id}\n作成日時: ${newNote.createdAt}`,
        },
      ],
    };
  }

  // ==================================================================
  // get_notes ツール実行処理：すべてのメモを取得
  // ==================================================================
  if (request.params.name === "get_notes") {
    // 既存のメモを読み込む
    const notes = readNotes();

    if (notes.length === 0) {
      return {
        content: [
          {
            type: "text",
            text: "メモがありません。",
          },
        ],
      };
    }

    // 各メモをフォーマット
    const notesText = notes.map(note => {
      const preview = note.content.substring(0, 100);
      const truncated = note.content.length > 100 ? "..." : "";
      return `ID: ${note.id}\n作成日時: ${note.createdAt}\n内容: ${preview}${truncated}`;
    }).join("\n---\n");

    return {
      content: [
        {
          type: "text",
          text: notesText,
        },
      ],
    };
  }

  // ==================================================================
  // find_note ツール実行処理：メモを検索
  // ==================================================================
  if (request.params.name === "find_note") {
    const { searchText } = request.params.arguments;

    // 既存のメモを読み込む
    const notes = readNotes();

    // 検索テキストを含むメモをフィルタリング
    const results = notes.filter(note =>
      note.content.toLowerCase().includes(searchText.toLowerCase())
    );

    if (results.length === 0) {
      return {
        content: [
          {
            type: "text",
            text: `「${searchText}」を含むメモは見つかりません。`,
          },
        ],
      };
    }

    // 結果をフォーマット
    const resultsText = results.map(note => {
      const preview = note.content.substring(0, 100);
      const truncated = note.content.length > 100 ? "..." : "";
      return `ID: ${note.id}\n作成日時: ${note.createdAt}\n内容: ${preview}${truncated}`;
    }).join("\n---\n");

    return {
      content: [
        {
          type: "text",
          text: `${results.length}件のメモが見つかりました。\n\n${resultsText}`,
        },
      ],
    };
  }

  // ==================================================================
  // delete_note ツール実行処理：メモを削除
  // ==================================================================
  if (request.params.name === "delete_note") {
    const { id } = request.params.arguments;

    // 既存のメモを読み込む
    const notes = readNotes();

    // 削除するメモを探す
    const noteIndex = notes.findIndex(note => note.id === id);
    if (noteIndex === -1) {
      return {
        content: [
          {
            type: "text",
            text: `ID ${id} のメモは見つかりません。`,
          },
        ],
      };
    }

    // メモを削除
    const deletedNote = notes.splice(noteIndex, 1)[0];

    // ファイルに書き込む
    writeNotes(notes);

    return {
      content: [
        {
          type: "text",
          text: `ID ${id} のメモを削除しました。\n削除されたメモ: ${deletedNote.content.substring(0, 50)}...`,
        },
      ],
    };
  }

  // 定義されていないツールが呼ばれた場合、エラーを返す
  throw new Error(`不明なツール: ${request.params.name}`);
